import requests
import json


wilayah_url = "https://ibnux.github.io/BMKG-importer/cuaca/wilayah.json"
response_wilayah = requests.get(wilayah_url)
response_wilayah_json = json.loads(response_wilayah.text)
print(response_wilayah_json)
